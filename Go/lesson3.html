Урок 3: Условия, циклы и функции - Go

Введение

Условия, циклы и функции в Go позволяют создавать структурированные программы. Условия управляют выполнением кода, циклы повторяют действия, а функции организуют код в переиспользуемые блоки. В этом уроке вы изучите эти концепции.

Условные операторы

Go использует if, else if и else для условий.


package main

import "fmt"

func main() {
    age := 20
    if age >= 18 {
        fmt.Println("Вы взрослый")
    } else if age >= 13 {
        fmt.Println("Вы подросток")
    } else {
        fmt.Println("Вы ребёнок")
    }
}

Циклы

Go имеет только один тип цикла — for, который может использоваться разными способами.

Классический for


package main

import "fmt"

func main() {
    for i := 1; i <= 5; i++ {
        fmt.Println("Итерация", i)
    }
}

Цикл как while


package main

import "fmt"

func main() {
    count := 1
    for count <= 5 {
        fmt.Println("Итерация", count)
        count++
    }
}

Цикл по слайсу


package main

import "fmt"

func main() {
    numbers := []int{1, 2, 3, 4, 5}
    for i, num := range numbers {
        fmt.Printf("Индекс: %d, Значение: %d\n", i, num)
    }
}

Функции

Функции в Go объявляются с помощью func. Они могут возвращать значения.


package main

import "fmt"

func greet(name string) string {
    return "Привет, " + name + "!"
}

func main() {
    fmt.Println(greet("Алиса")) // Привет, Алиса!
}

Функции с несколькими возвращаемыми значениями:


package main

import "fmt"

func addAndSubtract(a, b int) (int, int) {
    return a + b, a - b
}

func main() {
    sum, diff := addAndSubtract(10, 3)
    fmt.Println("Сумма:", sum, "Разность:", diff)
}
